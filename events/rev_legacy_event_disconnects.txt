add_namespace = usa

# Prevents pre-modern historical flavor from firing in late bookmarks.
# This overrides MD's usa.2 event trigger with a date gate.
country_event = {
	id = usa.2
	title = usa.2.t
	desc = usa.2.d
	picture = GFX_uss_cole_bombing
	fire_only_once = yes
	trigger = {
		tag = USA
		NOT = { has_country_flag = collapsed_nation }
		date < 2001.1.1
	}

	option = {
		name = usa.2.o1
		log = "[GetDateText]: [This.GetName]: usa.2.o1 executed"

		news_event = { hours = 6 id = usa_news.18 }
		add_named_threat = { threat = 0.5 name = usa.2.t }

		if = {
			limit = { country_exists = SUD }
			add_opinion_modifier = { target = SUD modifier = recent_actions_negative }
			reverse_add_opinion_modifier = { target = SUD modifier = recent_actions_negative }
		}
		else = {
			random_country = {
				limit = {
					has_government = fascism
					NOT = { tag = SAU }
					NOT = { tag = UAE }
					NOT = { tag = OMA }
					NOT = { tag = QAT }
					NOT = { tag = BHR }
					NOT = { tag = KUW }
					NOT = { tag = JOR }
				}
				add_opinion_modifier = { target = USA modifier = recent_actions_negative }
				reverse_add_opinion_modifier = { target = USA modifier = recent_actions_negative }
			}
		}
		USA_congress_small_opposition = yes
		set_country_flag = usa_uss_cole_attack

		ai_chance = {
			base = 1
		}
	}

	option = {
		name = usa.2.o2
		log = "[GetDateText]: [This.GetName]: usa.2.o2 executed"

		news_event = { hours = 6 id = usa_news.18 }
		navy_experience = 5
		add_named_threat = { threat = 0.5 name = usa.2.t }

		if = {
			limit = { country_exists = SUD }
			add_opinion_modifier = { target = SUD modifier = recent_actions_negative }
			reverse_add_opinion_modifier = { target = SUD modifier = recent_actions_negative }
			SUD = { country_event = { id = usa.2000 days = 7 random_hours = 2 } }
		}
		else = {
			random_country = {
				limit = {
					has_government = fascism
					NOT = { tag = SAU }
					NOT = { tag = UAE }
					NOT = { tag = OMA }
					NOT = { tag = QAT }
					NOT = { tag = BHR }
					NOT = { tag = KUW }
					NOT = { tag = JOR }
				}
				add_opinion_modifier = { target = USA modifier = recent_actions_negative }
				reverse_add_opinion_modifier = { target = USA modifier = recent_actions_negative }
				country_event = { id = usa.2000 days = 7 random_hours = 2 }
			}
		}
		USA_congress_small_opposition = yes
		set_country_flag = usa_uss_cole_attack

		ai_chance = {
			base = 0
			modifier = {
				add = 0.25
				is_historical_focus_on = no
			}
			modifier = {
				add = 0.25
				has_government = nationalist
				has_government = communism
			}
		}
	}
}
