on_actions = {
	on_daily = {
		effect = {
			if = {
				limit = {
					tag = CHI
					NOT = { has_country_flag = rev_chi_tw_april_2029_window_done }
					date > 2029.3.31
					date < 2029.4.5
					country_exists = USA
					country_exists = USB
					USA = { has_war_with = USB }
					country_exists = TAI
					NOT = { has_war_with = TAI }
				}
				country_event = { id = rev_chi_tw.1 }
			}

			# Keep the US civil-war actors out of the Taiwan war window.
			if = {
				limit = {
					has_global_flag = rev_chi_tw_non_intervention_window_active
					tag = USA
					country_exists = CHI
					has_war_with = CHI
				}
				white_peace = CHI
			}
			if = {
				limit = {
					has_global_flag = rev_chi_tw_non_intervention_window_active
					tag = USB
					country_exists = CHI
					has_war_with = CHI
				}
				white_peace = CHI
			}
			if = {
				limit = {
					has_global_flag = rev_chi_tw_non_intervention_window_active
					tag = HWI
					country_exists = CHI
					has_war_with = CHI
				}
				white_peace = CHI
			}

			if = {
				limit = {
					has_global_flag = rev_chi_tw_non_intervention_window_active
					NOT = { country_exists = TAI }
				}
				clr_global_flag = rev_chi_tw_non_intervention_window_active
			}
			if = {
				limit = {
					has_global_flag = rev_chi_tw_non_intervention_window_active
					country_exists = CHI
					CHI = { NOT = { has_war_with = TAI } }
				}
				clr_global_flag = rev_chi_tw_non_intervention_window_active
			}
		}
	}
}
