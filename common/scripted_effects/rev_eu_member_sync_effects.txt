rev_rebuild_eu_member_array = {
	# Rebuild from actual country ideas to avoid stale hard-coded startup lists.
	every_country = {
		remove_from_array = { global.EU_member = THIS }
	}

	every_country = {
		limit = { has_idea = EU_member }
		add_to_array = { global.EU_member = THIS }
	}

	# Keep aggregate count aligned with membership.
	set_variable = { var = global.var_EUmemberstatesTotal value = 0 }
	every_country = {
		limit = { has_idea = EU_member }
		add_to_variable = { var = global.var_EUmemberstatesTotal value = 1 }
	}
}
