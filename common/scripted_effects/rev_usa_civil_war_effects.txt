rev_begin_usa_usb_war = {
	# Core conflict branch: USB (Trump side) vs USA only.
	if = {
		limit = {
			USB = { exists = yes }
			USA = { exists = yes }
			USA = { NOT = { has_war_with = USB } }
		}
		USA = {
			declare_war_on = {
				target = USB
				type = annex_everything
			}
		}
	}

	# Keep Canada and NATO countries neutral in the civil war,
	# but strongly bias their AI toward supporting USA with volunteers.
	every_country = {
		if = {
			limit = {
				OR = {
					tag = CAN
					has_idea = NATO_member
				}
				NOT = { tag = USA }
				NOT = { tag = USB }
				exists = yes
			}
			if = {
				limit = { has_war_with = USA }
				white_peace = USA
			}
			if = {
				limit = { has_war_with = USB }
				white_peace = USB
			}
			if = {
				limit = { is_ai = yes }
				add_ai_strategy = {
					type = send_volunteers_desire
					id = USA
					value = 500
				}
				add_ai_strategy = {
					type = support
					id = USA
					value = 250
				}
				add_ai_strategy = {
					type = befriend
					id = USA
					value = 150
				}
				add_ai_strategy = {
					type = antagonize
					id = USB
					value = 120
				}
			}
		}
	}
}
