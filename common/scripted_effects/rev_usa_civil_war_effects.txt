rev_begin_usa_usb_war = {
# Core conflict branch: USB (Trump side) vs USA.
	if = {
		limit = {
			USB = { exists = yes }
			USA = { exists = yes }
			USA = { NOT = { has_war_with = USB } }
		}
		USA = {
			declare_war_on = {
				target = USB
				type = annex_everything
			}
		}
	}

	# Canada joins USA directly (if independent).
	if = {
		limit = {
			CAN = { exists = yes }
			CAN = { NOT = { is_subject = yes } }
			CAN = { NOT = { has_war_with = USA } }
			USA = { NOT = { has_war_with = CAN } }
		}
		CAN = {
			declare_war_on = {
				target = USB
				type = annex_everything
			}
		}
	}

	# Independent Cuba joins USB directly.
	if = {
		limit = {
			CUB = { exists = yes }
			CUB = { NOT = { is_subject = yes } }
			CUB = { NOT = { has_war_with = USB } }
			CUB = { NOT = { has_war_with = USA } }
		}
		CUB = {
			declare_war_on = {
				target = USA
				type = annex_everything
			}
		}
	}

	# NATO countries can volunteer for USA, but are not forced into the war.
	every_country = {
		limit = {
			has_idea = NATO_member
			NOT = { tag = USA }
			NOT = { tag = USB }
			exists = yes
		}
		add_ai_strategy = {
			type = send_volunteers_desire
			id = USA
			value = 300
		}
		add_ai_strategy = {
			type = support
			id = USA
			value = 200
		}
	}
}
