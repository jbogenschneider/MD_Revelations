rev_apply_nato_sov_embargo = {
	if = {
		limit = {
			SOV = {
				exists = yes
				has_war = yes
			}
		}
		SOV = {
			if = {
				limit = { NOT = { has_country_flag = SOV_western_sanctions } }
				set_country_flag = SOV_western_sanctions
				add_dynamic_modifier = { modifier = SOV_western_sanctions_modifier }
			}
		}
		every_country = {
			limit = {
				exists = yes
				has_idea = NATO_member
				NOT = { tag = SOV }
				NOT = { is_subject = yes }
			}
			if = {
				limit = { NOT = { has_country_flag = sanctions_against_SOV_flag } }
				set_country_flag = sanctions_against_SOV_flag
			}
			if = {
				limit = {
					NOT = { tag = USA }
				}
				add_opinion_modifier = { target = SOV modifier = minor_sanctions }
			}
			if = {
				limit = {
					tag = USA
				}
				add_opinion_modifier = { target = SOV modifier = sanctions }
			}
			if = {
				limit = {
					NOT = { is_embargoing = SOV }
				}
				send_embargo = SOV
			}
		}
	}
}

rev_remove_nato_sov_embargo = {
	if = {
		limit = {
			SOV = {
				exists = yes
				has_war = no
			}
		}
		SOV = {
			if = {
				limit = { has_country_flag = SOV_western_sanctions }
				clr_country_flag = SOV_western_sanctions
			}
		}
		every_country = {
			limit = {
				exists = yes
				has_idea = NATO_member
				NOT = { tag = SOV }
				NOT = { is_subject = yes }
			}
			if = {
				limit = {
					is_embargoing = SOV
				}
				break_embargo = SOV
			}
			if = {
				limit = { has_country_flag = sanctions_against_SOV_flag }
				clr_country_flag = sanctions_against_SOV_flag
			}
			remove_opinion_modifier = { target = SOV modifier = minor_sanctions }
			remove_opinion_modifier = { target = SOV modifier = sanctions }
		}
	}
}
